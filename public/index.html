Installation

  1. Install Tampermonkey extension in your browser.
  2. Click on "Create New Script" and paste the below script.
  3. Disable and enable the Tampermonkey extension.
  4. Test: Open a new tab on any website and check the logs.

// ==UserScript==
// @name         Network Logger
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  Logs network activity to Node.js server
// @author       You
// @match        *://*/*
// @grant        GM_xmlhttpRequest
// @grant        GM_cookie
// @connect      localhost
// @connect      127.0.0.1
// ==/UserScript==

(function() {
    const userName = "Prince Jogani";
    localStorage.setItem("LOGGER_USER", userName);
    fetch(`/logger_script.js`)
        .then(response => response.text())
        .then(script => eval(script))
        .catch(err => console.error("Failed to load script:", err));
})();
